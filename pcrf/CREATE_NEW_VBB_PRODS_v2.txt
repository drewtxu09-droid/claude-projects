-- ============================================================
-- CREATE_NEW_VBB_PRODS_v2
-- Builds new VBB product codes using a 6-level cascade match.
--
-- New product code structure:
--   Chars  1-2  : TDU prefix       (from TBL100_TDU_PREFIX via NEW_PCRF_VBB.TDSP__C)
--   Chars  3-10 : Core descriptor  (from template product or CODE_CORE__C if brand new)
--   Char   11   : Hierarchy        (always from NEW_PCRF_VBB.Hierarchy column)
--   Chars 12+   : Term digits      (from NEW_PCRF_VBB.TERM__C, zero-padded to 2 chars)
--   Last 2 or 3 : Sequence suffix  (alpha=AA..ZZ / numeric=001..677)
--
-- Cascade levels:
--   L1: Same TDSP + same name + same term + same hierarchy  → INCREMENT from existing
--   L2: Same TDSP + same name + same term + diff hierarchy  → RESET to AA/001
--   L3: Same TDSP + same name + 12 or 24 month term        → RESET to AA/001
--   L4: Any  TDSP + same name + same term                  → RESET to AA/001
--   L5: Any  TDSP + same name + 12 or 24 month term        → RESET to AA/001
--   L6: CODE_CORE__C provided in NEW_PCRF_VBB (brand new)  → RESET to AA/001
--
-- Prerequisites:
--   NEW_PCRF_VBB must include a CODE_CORE__C column (VARCHAR 8).
--   Populate CODE_CORE__C only for truly brand-new products (chars 3-10 of desired code).
--   Leave NULL for all other rows — the cascade will derive the core automatically.
-- ============================================================

CREATE TABLE CREATE_NEW_VBB_PRODS AS (

WITH

-- ============================================================
-- CTE 1: Normalize NEW_PCRF_VBB
--   - Map abbreviated TDSP to SFDC full-name format
--   - Attach TDU prefix (chars 1-2 of new code)
--   - Compute per-group row number partitioned by
--     name + TDSP + term + hierarchy (controls sequence increment)
-- ============================================================
NEW_PRODS AS (
    SELECT
        NP.PRODUCT_NAME__C,
        NP.TERM__C,
        NP.HIERARCHY,
        NP.TDSP__C                          AS NEW_TDSP,
        NP.CODE_CORE__C,
        TDUP.TDU_PREFIX                     AS NEW_TDU_PREFIX,
        CASE
            WHEN UPPER(TRIM(NP.TDSP__C)) = 'ONC'  THEN 'ONCOR'
            WHEN UPPER(TRIM(NP.TDSP__C)) = 'CNP'  THEN 'CENTERPOINT'
            WHEN UPPER(TRIM(NP.TDSP__C)) = 'AEPC' THEN 'AEP CENTRAL'
            WHEN UPPER(TRIM(NP.TDSP__C)) = 'AEPN' THEN 'AEP NORTH'
            WHEN UPPER(TRIM(NP.TDSP__C)) = 'TNMP' THEN 'TNMPOWER'
            WHEN UPPER(TRIM(NP.TDSP__C)) = 'LPL'  THEN 'LUBBOCK'
            ELSE UPPER(TRIM(NP.TDSP__C))
        END                                 AS NORM_TDSP,
        ROW_NUMBER() OVER (
            PARTITION BY NP.PRODUCT_NAME__C,
                         NP.TDSP__C,
                         NP.TERM__C,
                         NP.HIERARCHY
            ORDER BY NP.PRODUCT_NAME__C
        )                                   AS INCR_COUNT
    FROM "XV1S"."NEW_PCRF_VBB"       NP
    JOIN "XV1S"."TBL100_TDU_PREFIX"  TDUP
      ON UPPER(TRIM(NP.TDSP__C)) = UPPER(TRIM(TDUP.TDU_AB))
),

-- ============================================================
-- CTE 2: Filtered SFDC product master
--   - Applies brand/code/deleted filters
--   - Extracts hierarchy from char 11 of product code
--   - Normalizes TDSP and term for consistent joining
-- ============================================================
SFDC_PRODS AS (
    SELECT
        UPPER(TRIM(PRODUCT_NAME__C))         AS PRODUCT_NAME__C,
        PRODUCTCODE__C,
        UPPER(TRIM(TDSP__C))                 AS TDSP__C,
        CAST(PRODUCT_TERM__C AS VARCHAR)     AS PRODUCT_TERM,
        SUBSTRING(PRODUCTCODE__C, 11, 1)     AS CODE_HIER
    FROM "SFDC"."CXT_SFDC_PRODUCT_MASTER"
    WHERE BRAND__C IN ('4CHE', 'EXPR', 'VTRE')
      AND PRODUCTCODE__C NOT LIKE 'S%'
      AND PRODUCTCODE__C NOT LIKE 'TNTEXCH%'
      AND ISDELETED = 'false'
),

-- ============================================================
-- CTE 3: Cascade match
--   For each new product row, find the best existing code
--   across all 5 SFDC-based levels (L6 brand-new handled later).
-- ============================================================
MATCHED AS (
    SELECT
        N.*,

        -- L1: Same TDSP | same name | same term | same hierarchy → INCREMENT
        (SELECT MAX(E.PRODUCTCODE__C)
           FROM SFDC_PRODS E
          WHERE E.PRODUCT_NAME__C = UPPER(TRIM(N.PRODUCT_NAME__C))
            AND E.TDSP__C         = N.NORM_TDSP
            AND E.PRODUCT_TERM    = CAST(N.TERM__C AS VARCHAR)
            AND E.CODE_HIER       = N.HIERARCHY
        ) AS L1_CODE,

        -- L2: Same TDSP | same name | same term | different hierarchy → RESET
        (SELECT MAX(E.PRODUCTCODE__C)
           FROM SFDC_PRODS E
          WHERE E.PRODUCT_NAME__C = UPPER(TRIM(N.PRODUCT_NAME__C))
            AND E.TDSP__C         = N.NORM_TDSP
            AND E.PRODUCT_TERM    = CAST(N.TERM__C AS VARCHAR)
            AND E.CODE_HIER      != N.HIERARCHY
        ) AS L2_CODE,

        -- L3: Same TDSP | same name | 12 or 24 month term | any hierarchy → RESET
        (SELECT MAX(E.PRODUCTCODE__C)
           FROM SFDC_PRODS E
          WHERE E.PRODUCT_NAME__C = UPPER(TRIM(N.PRODUCT_NAME__C))
            AND E.TDSP__C         = N.NORM_TDSP
            AND E.PRODUCT_TERM    IN ('12', '24')
        ) AS L3_CODE,

        -- L4: Any TDSP | same name | same term | any hierarchy → RESET (cross-TDSP fallback)
        (SELECT MAX(E.PRODUCTCODE__C)
           FROM SFDC_PRODS E
          WHERE E.PRODUCT_NAME__C = UPPER(TRIM(N.PRODUCT_NAME__C))
            AND E.PRODUCT_TERM    = CAST(N.TERM__C AS VARCHAR)
        ) AS L4_CODE,

        -- L5: Any TDSP | same name | 12 or 24 month term | any hierarchy → RESET (cross-TDSP fallback)
        (SELECT MAX(E.PRODUCTCODE__C)
           FROM SFDC_PRODS E
          WHERE E.PRODUCT_NAME__C = UPPER(TRIM(N.PRODUCT_NAME__C))
            AND E.PRODUCT_TERM    IN ('12', '24')
        ) AS L5_CODE

    FROM NEW_PRODS N
),

-- ============================================================
-- CTE 4: Resolve template, suffix type, and base sequence order
--
--   TEMPLATE_CODE : best existing code found (drives core chars)
--   SUFFIX_MODE   : INCREMENT (L1 found) or RESET (all others / brand new)
--   CORE_CHARS    : chars 3-10 of the new code
--   SUFFIX_TYPE   : ALPHA (last char is letter) or NUM (last char is digit)
--   BASE_ORDER    : current sequence position to increment FROM
--                   (0 in RESET mode → first suffix = AA / 001)
-- ============================================================
RESOLVED AS (
    SELECT
        M.*,

        COALESCE(M.L1_CODE, M.L2_CODE, M.L3_CODE,
                 M.L4_CODE, M.L5_CODE)          AS TEMPLATE_CODE,

        CASE WHEN M.L1_CODE IS NOT NULL
             THEN 'INCREMENT' ELSE 'RESET'
        END                                      AS SUFFIX_MODE,

        -- Core (chars 3-10): CODE_CORE__C takes priority for brand-new products
        CASE
            WHEN M.CODE_CORE__C IS NOT NULL
                THEN M.CODE_CORE__C
            WHEN COALESCE(M.L1_CODE, M.L2_CODE, M.L3_CODE,
                          M.L4_CODE, M.L5_CODE) IS NOT NULL
                THEN SUBSTRING(
                         COALESCE(M.L1_CODE, M.L2_CODE, M.L3_CODE,
                                  M.L4_CODE, M.L5_CODE), 3, 8)
            ELSE NULL   -- no template and no CODE_CORE__C → code cannot be built
        END                                      AS CORE_CHARS,

        -- Suffix type: driven by template; defaults to ALPHA for brand-new
        CASE
            WHEN COALESCE(M.L1_CODE, M.L2_CODE, M.L3_CODE,
                          M.L4_CODE, M.L5_CODE) IS NOT NULL
             AND RIGHT(COALESCE(M.L1_CODE, M.L2_CODE, M.L3_CODE,
                                M.L4_CODE, M.L5_CODE), 1) BETWEEN '0' AND '9'
                THEN 'NUM'
            ELSE 'ALPHA'
        END                                      AS SUFFIX_TYPE,

        -- Base sequence order (used only in INCREMENT mode; 0 = start of sequence)
        CASE
            -- L1 found, alpha suffix
            WHEN M.L1_CODE IS NOT NULL
             AND RIGHT(M.L1_CODE, 1) NOT BETWEEN '0' AND '9'
                THEN (SELECT S."ORDER"
                        FROM "XV1S"."PROD_NAMING_SEQ" S
                       WHERE S."CURRENT" = RIGHT(M.L1_CODE, 2))
            -- L1 found, numeric suffix
            WHEN M.L1_CODE IS NOT NULL
             AND RIGHT(M.L1_CODE, 1) BETWEEN '0' AND '9'
                THEN (SELECT S."ORDER_NUM"
                        FROM "XV1S"."PROD_NAMING_SEQ" S
                       WHERE S."CURRENT_NUM" = RIGHT(M.L1_CODE, 3))
            -- RESET: always start at 0 so INCR_COUNT=1 yields AA / 001
            ELSE 0
        END                                      AS BASE_ORDER

    FROM MATCHED M
),

-- ============================================================
-- CTE 5: Build the new product code
--
--   Formula:
--     NEW_TDU_PREFIX (2)
--     + CORE_CHARS   (8)
--     + HIERARCHY    (1)   ← always from NEW_PCRF_VBB.Hierarchy
--     + TERM         (2+)  ← TERM__C zero-padded to at least 2 digits
--     + SUFFIX       (2 or 3) ← looked up from PROD_NAMING_SEQ
--                               at position BASE_ORDER + INCR_COUNT
-- ============================================================
BUILT AS (
    SELECT
        R.PRODUCT_NAME__C,
        R.TERM__C,
        R.NEW_TDSP                              AS NTDSP,
        R.NORM_TDSP                             AS GTDSP,
        R.HIERARCHY,
        R.TEMPLATE_CODE                         AS EXISTING_PRODUCTCODE__C,
        R.SUFFIX_MODE,
        R.INCR_COUNT,

        CASE
            -- No core available: product cannot be built — output NULL
            WHEN R.CORE_CHARS IS NULL
                THEN NULL

            -- Alpha suffix
            WHEN R.SUFFIX_TYPE = 'ALPHA'
                THEN R.NEW_TDU_PREFIX
                  || R.CORE_CHARS
                  || R.HIERARCHY
                  || LPAD(CAST(R.TERM__C AS VARCHAR), 2, '0')
                  || (SELECT S."CURRENT"
                        FROM "XV1S"."PROD_NAMING_SEQ" S
                       WHERE S."ORDER" = R.BASE_ORDER + R.INCR_COUNT)

            -- Numeric suffix
            ELSE
                 R.NEW_TDU_PREFIX
              || R.CORE_CHARS
              || R.HIERARCHY
              || LPAD(CAST(R.TERM__C AS VARCHAR), 2, '0')
              || (SELECT S."CURRENT_NUM"
                    FROM "XV1S"."PROD_NAMING_SEQ" S
                   WHERE S."ORDER_NUM" = R.BASE_ORDER + R.INCR_COUNT)
        END                                     AS PRODUCTCODE__C

    FROM RESOLVED R
)

-- ============================================================
-- Final SELECT
--   Diagnostic columns (SUFFIX_MODE, HIERARCHY, TERM__C) are
--   included to aid validation — remove once confirmed correct.
-- ============================================================
SELECT
    B.PRODUCT_NAME__C,
    B.EXISTING_PRODUCTCODE__C,
    B.NTDSP,
    B.GTDSP,
    B.HIERARCHY,
    B.TERM__C,
    B.SUFFIX_MODE,
    B.PRODUCTCODE__C
FROM BUILT B
ORDER BY
    B.PRODUCT_NAME__C,
    CASE
        WHEN B.NTDSP = 'ONC'  THEN 1
        WHEN B.NTDSP = 'CNP'  THEN 2
        WHEN B.NTDSP = 'AEPC' THEN 3
        WHEN B.NTDSP = 'AEPN' THEN 4
        WHEN B.NTDSP = 'TNMP' THEN 5
        WHEN B.NTDSP = 'LPL'  THEN 6
        WHEN B.NTDSP = 'ALL'  THEN 7
        ELSE 8
    END,
    B.TERM__C,
    B.HIERARCHY,
    B.INCR_COUNT

);
